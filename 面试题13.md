面试题[13](https://leetcode-cn.com/problems/ji-qi-ren-de-yun-dong-fan-wei-lcof/)

* 搜索要判断是不是需要向下，上，左右四个方向搜索。
* 这种题，多少点能到达用搜索，到达某一点有多少种方法可以使用DP.S

```java
class Solution {
    int m;
    int n;
    int k;
    int count =0;
    public int movingCount(int m, int n, int k) {
        this.m=m;
        this.n=n;
        this.k=k;
        boolean[][] visited = new boolean[m][n];
        DFS(0,0,visited);
        return count;
    }
    public void DFS(int i,int j,boolean[][] visited){
        if(check(i,j,this.k)){
             if(visited[i][j]){
                return ;
            }
            count++;
            visited[i][j]=true;
            DFS(i+1,j,visited);
            DFS(i,j+1,visited);
        }else{
            return ;
        }
    }
    public boolean check(int i,int j ,int k){
        if(i<0 || i>=m || j<0 || j>= n){
            return false;
        }
        return helper(i)+helper(j)<=k;
    }
    public int helper(int i){
        int s = 0;
        while(i>0){
            s+=i%10;
            i/=10;
        }
        return s;
    }
}
```
