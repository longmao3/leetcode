面试题[12](https://leetcode-cn.com/problems/ju-zhen-zhong-de-lu-jing-lcof/)

```java
class Solution {
    boolean[][] flag;
    int len,len_2;
    char[][] board;
    char[] chars;
     int[][] dir;
    public boolean exist(char[][] board, String word) {
        if (board == null || board[0] == null || board.length == 0 || board[0].length == 0 || word             == null || word.equals("")) {
            return false;
        }
        
        len = board.length;
        len_2=board[0].length;
        flag = new boolean[len][len_2];
        this.board = board;
        this.chars = word.toCharArray();
        dir = new int[][]{{-1,0},{1,0},{0,-1},{0,1}};
    
        for(int i=0;i<len;i++){
            for(int j=0;j<len_2;j++){
                if(dfs(i,j,0)){
                    return true;
                }
            }
        }
        return false;
    }

    public boolean dfs(int i,int j,int index){
        if( !(i>=0 && i<len && j>=0 && j<len_2)) return false;
        if(flag[i][j]){
            return false;
        }
        if(board[i][j] != chars[index]){
            return false;
        }
        flag[i][j] = true;
        if(index+1 == chars.length){
            return true;
        }
        for(int d=0;d<dir.length;d++){
            int new_x = i+dir[d][0];
            int new_y = j+dir[d][1];
            if(dfs(new_x,new_y,index+1)){
                return true;
            }
        }
        flag[i][j] = false;
        return false;
    }
}
```
