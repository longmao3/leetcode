class Solution {
    public int[] reversePrint(ListNode head) {
        int count=0;
        ListNode cur = head;
        while(cur != null){
            cur = cur.next;
            count++;
        }
        int[]  res = new int[count];
        for(int i=count-1;i>=0;i--){
            res[i] = head.val;
            head = head.next;
        }
        return res;
    }
}
